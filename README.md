Try - Catch
```java
try{
    Alumno alumno = new Alumno();
    alumno.setName("Fernando");
    alumno.setAge(31);
    alumno.setMatricula(26568);
    alumno.getAge();
    alumno.getMatricula();
    alumno.getName();
```
